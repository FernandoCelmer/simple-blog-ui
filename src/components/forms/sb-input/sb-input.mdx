# Input

Single-line text input with built-in label, hint, and error states. Emits `sb-input` (on every keystroke) and `sb-change` (on blur/enter). Supports all native input types.

## Usage

<FrameworkTabs
  html={`<sb-input
  label="Email"
  type="email"
  placeholder="you@example.com"
  hint="We never share your email."
  required
></sb-input>

<script>
  document.querySelector('sb-input')
    .addEventListener('sb-input', (e) => console.log(e.detail.value));
</script>`}
  react={`import { Input } from 'simple-blog-ui/react';
import { useState } from 'react';

function EmailField() {
  const [email, setEmail] = useState('');
  return (
    <Input
      label="Email"
      type="email"
      placeholder="you@example.com"
      hint="We never share your email."
      required
      value={email}
      onInput={(e) => setEmail(e.detail.value)}
    />
  );
}`}
  vue={`<script setup>
import { ref } from 'vue';
const email = ref('');
</script>

<template>
  <sb-input
    label="Email"
    type="email"
    placeholder="you@example.com"
    hint="We never share your email."
    required
    :value="email"
    @sb-input="email = $event.detail.value"
  />
</template>`}
/>
