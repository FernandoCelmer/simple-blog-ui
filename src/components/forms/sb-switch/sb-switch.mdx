# Switch

Sliding on/off toggle with a rectangular track and square thumb. Same API as checkbox — emits `sb-change` with `{ checked, value }`.

## Usage

<FrameworkTabs
  html={`<sb-switch name="dark-mode">Dark mode</sb-switch>

<script>
  document.querySelector('sb-switch')
    .addEventListener('sb-change', (e) => console.log(e.detail.checked));
</script>`}
  react={`import { Switch } from 'simple-blog-ui/react';
import { useState } from 'react';

function DarkModeToggle() {
  const [on, setOn] = useState(false);
  return (
    <Switch
      checked={on}
      onChange={(e) => setOn(e.detail.checked)}
    >
      Dark mode
    </Switch>
  );
}`}
  vue={`<script setup>
import { ref } from 'vue';
const darkMode = ref(false);
</script>

<template>
  <sb-switch
    :checked="darkMode"
    @sb-change="darkMode = $event.detail.checked"
  >
    Dark mode
  </sb-switch>
</template>`}
/>
