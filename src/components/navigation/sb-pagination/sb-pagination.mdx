# Pagination

Numbered page selector with prev/next arrows. Props: `page`, `total`, `siblings` (how many pages to show around the current one). Emits `sb-page-change`.

## Usage

<FrameworkTabs
  html={`<sb-pagination page="6" total="20" siblings="1"></sb-pagination>

<script>
  document.querySelector('sb-pagination')
    .addEventListener('sb-page-change', (e) => console.log(e.detail.page));
</script>`}
  react={`import { Pagination } from 'simple-blog-ui/react';
import { useState } from 'react';

function List() {
  const [page, setPage] = useState(1);
  return (
    <Pagination
      page={page}
      total={20}
      siblings={1}
      onPageChange={(e) => setPage(e.detail.page)}
    />
  );
}`}
  vue={`<script setup>
import { ref } from 'vue';
const page = ref(1);
</script>

<template>
  <sb-pagination
    :page="page"
    :total="20"
    :siblings="1"
    @sb-page-change="page = $event.detail.page"
  />
</template>`}
/>
