# Navbar

Top navigation bar with a `brand` slot and default slot for links. Collapses to a "Menu" toggle on viewports < 768px.

## Usage

<FrameworkTabs
  html={`<sb-navbar>
  <a slot="brand" href="/">Simple Blog</a>
  <sb-link variant="nav" href="/">Home</sb-link>
  <sb-link variant="nav" href="/posts">Posts</sb-link>
  <sb-link variant="nav" href="/about">About</sb-link>
</sb-navbar>`}
  react={`import { Navbar, Link } from 'simple-blog-ui/react';

<Navbar>
  <a slot="brand" href="/">Simple Blog</a>
  <Link variant="nav" href="/">Home</Link>
  <Link variant="nav" href="/posts">Posts</Link>
  <Link variant="nav" href="/about">About</Link>
</Navbar>`}
  vue={`<sb-navbar>
  <a slot="brand" href="/">Simple Blog</a>
  <sb-link variant="nav" href="/">Home</sb-link>
  <sb-link variant="nav" href="/posts">Posts</sb-link>
  <sb-link variant="nav" href="/about">About</sb-link>
</sb-navbar>`}
/>
