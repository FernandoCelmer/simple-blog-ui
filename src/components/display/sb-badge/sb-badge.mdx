# Badge

Uppercase tag-like label. Variants: `solid`, `outline`, `success`, `warning`, `danger`, `info`.

## Usage

<FrameworkTabs
  html={`<sb-badge variant="solid">New</sb-badge>
<sb-badge variant="success">Done</sb-badge>
<sb-badge variant="danger">Error</sb-badge>`}
  react={`import { Badge } from 'simple-blog-ui/react';

<Badge variant="solid">New</Badge>
<Badge variant="success">Done</Badge>
<Badge variant="danger">Error</Badge>`}
  vue={`<sb-badge variant="solid">New</sb-badge>
<sb-badge variant="success">Done</sb-badge>
<sb-badge variant="danger">Error</sb-badge>`}
/>
