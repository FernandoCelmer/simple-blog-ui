# Tag

Small bordered chip. Add `removable` for a × button that emits `sb-remove`.

## Usage

<FrameworkTabs
  html={`<sb-tag>javascript</sb-tag>
<sb-tag removable>lit</sb-tag>

<script>
  document.querySelector('sb-tag[removable]')
    .addEventListener('sb-remove', (e) => e.target.remove());
</script>`}
  react={`import { Tag } from 'simple-blog-ui/react';

<Tag>javascript</Tag>
<Tag removable onRemove={() => console.log('removed')}>lit</Tag>`}
  vue={`<sb-tag>javascript</sb-tag>
<sb-tag removable @sb-remove="onRemove">lit</sb-tag>`}
/>
