# Button

Monospace, uppercase, 2px solid border. Four variants (`primary`, `secondary`, `ghost`, `danger`) and three sizes (`sm`, `md`, `lg`). Emits a `sb-click` custom event when clicked and not disabled.

## Usage

<FrameworkTabs
  html={`<script type="module">
  import 'simple-blog-ui';
</script>

<sb-button variant="primary">Click me</sb-button>

<script>
  document.querySelector('sb-button')
    .addEventListener('sb-click', () => console.log('clicked'));
</script>`}
  react={`import { Button } from 'simple-blog-ui/react';
import 'simple-blog-ui/style.css';

export default function App() {
  return (
    <Button
      variant="primary"
      size="md"
      onClick={() => console.log('clicked')}
    >
      Click me
    </Button>
  );
}`}
  vue={`<script setup>
import 'simple-blog-ui';
import 'simple-blog-ui/style.css';
</script>

<template>
  <sb-button
    variant="primary"
    size="md"
    @sb-click="() => console.log('clicked')"
  >
    Click me
  </sb-button>
</template>`}
/>
