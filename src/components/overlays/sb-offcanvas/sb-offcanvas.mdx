# Offcanvas

Slide-in drawer from any edge (`right`, `left`, `top`, `bottom`). Trigger via `show()`, close with `close()` or Escape. Includes backdrop.

## Usage

<FrameworkTabs
  html={`<sb-button id="open">Filters</sb-button>
<sb-offcanvas id="oc" placement="right" title="Filters">
  <p>Filter options go here.</p>
</sb-offcanvas>

<script>
  const oc = document.getElementById('oc');
  document.getElementById('open').addEventListener('sb-click', () => oc.show());
</script>`}
  react={`import { Offcanvas, Button } from 'simple-blog-ui/react';
import { useRef } from 'react';

function Example() {
  const oc = useRef(null);
  return (
    <>
      <Button onClick={() => oc.current.show()}>Filters</Button>
      <Offcanvas ref={oc} placement="right" title="Filters">
        <p>Filter options go here.</p>
      </Offcanvas>
    </>
  );
}`}
  vue={`<script setup>
import { ref } from 'vue';
const oc = ref(null);
</script>

<template>
  <sb-button @sb-click="oc.show()">Filters</sb-button>
  <sb-offcanvas ref="oc" placement="right" title="Filters">
    <p>Filter options go here.</p>
  </sb-offcanvas>
</template>`}
/>
