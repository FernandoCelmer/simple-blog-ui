# Toast

Ephemeral notification anchored to a corner of the viewport. Variants (`default`, `success`, `warning`, `danger`, `info`) and four placements. Trigger imperatively via `show()`; auto-dismiss with `duration` (ms).

## Usage

<FrameworkTabs
  html={`<sb-toast id="t" title="Saved" variant="success" duration="3000">
  Your changes were saved.
</sb-toast>

<script>
  document.getElementById('t').show();
</script>`}
  react={`import { Toast } from 'simple-blog-ui/react';
import { useRef } from 'react';

function Example() {
  const toast = useRef(null);
  return (
    <>
      <button onClick={() => toast.current.show()}>Save</button>
      <Toast ref={toast} title="Saved" variant="success" duration={3000}>
        Your changes were saved.
      </Toast>
    </>
  );
}`}
  vue={`<script setup>
import { ref } from 'vue';
const toast = ref(null);
</script>

<template>
  <button @click="toast.show()">Save</button>
  <sb-toast ref="toast" title="Saved" variant="success" duration="3000">
    Your changes were saved.
  </sb-toast>
</template>`}
/>
