# Progress

Horizontal progress bar. Props: `value`, `max` (default 100), `variant` (default, success, warning, danger, info), `striped`, `show-label`.

## Usage

<FrameworkTabs
  html={`<sb-progress value="60" max="100"></sb-progress>

<sb-progress value="80" variant="success" striped show-label label="Upload"></sb-progress>`}
  react={`import { Progress } from 'simple-blog-ui/react';

<Progress value={60} max={100} />

<Progress value={80} variant="success" striped showLabel label="Upload" />`}
  vue={`<sb-progress :value="60" :max="100"></sb-progress>

<sb-progress :value="80" variant="success" striped show-label label="Upload"></sb-progress>`}
/>
